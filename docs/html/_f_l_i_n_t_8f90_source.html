62!! \section Installation
63!! FLINT is tested with ifort (18.0.2) compiler from Intel Parallel Studio XE Composer for Windows
64!! 2016 integrated with Microsoft Visual Studio Community version 2017. Some testing is also done
65!! with MinGW-W64 gfortran (gcc 8.1.0). Doxyfile is provided for generating extensive API
66!! documentation using Doxygen. FLINT has no dependency on any other library. The CMakeLists file
67!! is provided to generate Visual Studio projects or makefiles on Windows and Linux using cmake.
68!! Additionally, it generates cmake config files to easily link FLINT using the find_package()
69!! command. The steps to link FLINT in cmake-based projects are:
70!!
71!! + In cmake GUI or command-line, set FLINT_INSTALL_LIB_DIR to the desired directory,
72!! where the compiled library, modules, and cmake config files will be installed.
73!! + In cmake GUI or command-line, set FLINT_INSTALL_BIN_DIR to the desired directory,
74!! where the compiled test executables of FLINT will be installed.
75!! + In your project CMakeLists.txt, insert
76!!
77!! \code
78!! find_package(FLINT REQUIRED 0.9 CONFIG
79!! PATHS "<SAME_PATH_AS_IN_FLINT_INSTALL_LIB_DIR>")
80!! target_link_libraries(<YOUR_TARGET_NAME> FLINT::FLINT)
81!! \endcode

84!! \section usagesec How to Use
85!! See the test program files, test.f90 and DiffEq.f90, in tests folder for a comparatively complex
86!! example problem that uses multiple events. Following are the steps in brief for a simpler problem:
87!!
88!! + Create a differential equation system class by providing differential equation
89!! function, events function (if any), and parameters (if any).
90!!
91!! \code{.f90}
92!! use FLINT
93!! implicit none
94!!
95!! type, extends(DiffEqSys) :: TBSys
96!! real(WP) :: GM = 398600.436233_WP ! parameter constant
97!! contains
98!! procedure :: F => TwoBodyDE ! Differential equations
99!! procedure :: G => SampleEventTB ! events, use only if needed
100!! end type TBSys
101!!
102!! contains
103!!
104!! pure function TwoBodyDE(me, X, Y, Params)
105!! implicit none
106!! class(TBSys), intent(in) :: me !< Differential Equation object
107!! real(WP), intent(in) :: X
108!! real(WP), intent(in), dimension(:) :: Y
109!! real(WP), intent(in), dimension(:), optional :: Params
110!! real(WP), dimension(size(Y)) :: TwoBodyDE
111!!
112!! TwoBodyDE(1:3) = Y(4:6)
113!! TwoBodyDE(4:6) = -me%GM/(norm2(Y(1:3))**3)*Y(1:3) ! Two-body orbit diffeq
114!! end function TwoBodyDE
115!!
116!! subroutine SampleEventTB(me, X, Y, EvalEvents, Value, Direction, LocEvent, LocEventAction)
117!! implicit none
118!! class(TBSys), intent(in) :: me !< Differential Equation object
119!! real(WP), intent(in) :: X
120!! real(WP), dimension(:), intent(inout) :: Y
121!! integer, dimension(:), intent(in) :: EvalEvents
122!! real(WP), dimension(:), intent(out) :: Value
123!! integer, dimension(:), intent(out) :: Direction
124!! integer, intent(in), optional :: LocEvent
125!! integer(kind(FLINT_EVENTACTION_CONTINUE)), intent(out), optional :: LocEventAction
126!!
127!! if (EvalEvents(1)==1) Value(1) = Y(2) ! detect y-crossing
128!! if (EvalEvents(2)==1) Value(2) = Y(1) ! detect x-crossing
129!!
130!! Direction(1) = 1 ! detect -ve to +ve transitions for y coordinate
131!! Direction(2) = -1 ! detect +ve to -ve transitions for x coordinate
132!!
133!! ! Set actions for each event if located
134!! if (present(LocEvent) .AND. present(LocEventAction)) then
135!! if (LocEvent == 1) then
136!! LocEventAction = FLINT_EVENTACTION_CONTINUE
137!! else if (LocEvent == 2) then
138!! ! Mask the event-2 after first trigger
139!! LocEventAction = IOR(FLINT_EVENTACTION_CONTINUE, &
140!! FLINT_EVENTACTION_MASK)
141!! end if
142!! end if
143!! end subroutine SampleEventTB
144!! \endcode
145!!
146!! + Initialize the differential equation and ERK class objects for using Runge-Kutta integrators
147!!
148!! \code{.f90}
149!! use FLINT
150!! type(ERK_class) :: erk
151!! type(TBSys) :: diffeq
152!! diffeq%n = 6 ! Number of 1st-order differential equations
153!! diffeq%m = 2 ! Number of events
154!!
155!! ! Initialize the ERK object for 10000 max steps, DOP54 method with abs. tol. 1e-12, rel. tol. 1e-9
156!! ! and turn on interpolation and events detection
157!! call erk%Init(diffeq, 10000, Method=ERK_DOP54, ATol=[1e-12], RTol=[1e-9],&
158!! InterpOn=.TRUE.,EventsOn=.TRUE.)
159!! \endcode
160!!
161!! + Call the Integrate subroutine for performing the integration if Init was successful.
162!! Note if interpolation is enabled, then the IntStepsOn option for computing the states
163!! at the integrator's natural steps must not be set to True.
164!!
165!! \code{.f90}
166!! integer :: stiffstatus
167!! real(WP) :: x0, xf
168!! real(WP), dimension(6) :: y0, yf
169!! real(WP), allocatable, dimension(:,:) :: EventData ! allocated by Integrate
170!!
171!! stiffstatus = 1 ! detect stiffness and terminate if equations are stiff
172!! x0 = 0.0
173!! y0 = [6400.0_wp,0.0_wp,0.0_WP, 0.0_WP,7.69202528825512_WP,7.69202528825512_WP]
174!! xf = 161131.68239305308_WP
175!!
176!! ! Call Intgerate with final solution in yf, no initial step-size given, events-related
177!! ! data (event-id, x value, y state) in EventData, and all events are detected
178!! if (erk%status == FLINT_SUCCESS) then
179!! call erk%Integrate(x0, y0, xf, yf, StepSz=0.0, IntStepsOn=.FALSE.,&
180!! EventStates=EventData, EventMask = [.TRUE.,.TRUE.],StiffTest=stiffstatus)
181!! end if
182!! \endcode
183!!
184!! + Call the Interpolate function for computing solution on the desired grid of x values.
185!! The last parameter must be specified as True if user wants FLINT to keep the internal storage
186!! for calling Interpolate again. Otherwise, the internal storage is deleted and the user
187!! must integrate the equations again before calling Interpolate.
188!!
189!! \code{.f90}
190!! real(WP), dimension(:), allocatable :: Xarr1, Xarr2
191!! real(WP), dimension(6,:) :: Yarr1, Yarr2
192!!
193!! Xarr1 = [(x0 + (xf-x0)/10*i, i=0,9)] ! grid-1 with 10 points
194!! Xarr2 = [(x0 + (xf-x0)/1000*i, i=0,999)] ! grid-2 with 1000 points
195!! allocate(Yarr1(6,size(Xarr1))) ! allocate solution storage
196!! allocate(Yarr2(6,size(Xarr2)))
197!!
198!! if (erk%status == FLINT_SUCCESS) then
199!! ! interpolate and keep the internal storage for further calls
200!! call erk%Interpolate(Xarr1, Yarr1, .TRUE.)
201!! ! After this interpolation, delete the internal storage (default).
202!! call erk%Interpolate(Xarr2, Yarr2, .FALSE.)
203!! end if
204!!
205!! ! print the solutions
206!! print *, 'Solution at grid-1'
207!! print *, Yarr1
208!! print *, 'Solution at grid-2'
209!! print *, Yarr2
210!! ! print the event data
211!! print *, EventData(1,:) ! Time at which events occured
212!! print *, EventData(2:7,:) ! corresponding position and velocity states
213!! print *, Eventdata(8,:) ! Event-ID number
214!! \endcode
215!!
216!! For all the FLINT status codes and options supported by Init, Integrate, and Interpolate
217!! procedures along with the interfaces for user-supplied functions, see the FLINT_base
218!! module in FLINT_base.f90 file.
219!!
220!! \section testsec Testing
221!! The following tests were performed using the very first verion of FLINT and they
222!! are not updated for the latest version. In the tests, an orbit is propagated for
223!! 4 orbital periods. The propagation is repeated 5000 times. The tables give the
224!! total time for 5000 propagations and all the other testing parameters are the
225!! values for 1 propagation. The relative tolerance is taken as 1e-11 and absolute
226!! tolerance as 1e-14 in all the cases except when explicitly mentioned otherwise.
227!! Interpolation is used to capture 1000 points per orbit period, however delayed
228!! interpolation feature of FLINT is not used in these results.
229!!
